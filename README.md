# To connect to Mongo Atlas Server
setup and create your .env in the project directory

ATLAS_URI=mongodb+srv://[username]:[password]@[clusterName].xxxxxxx.mongodb.net/?retryWrites=true&w=majority

# Run the server
cd backend
run `node server.js`
